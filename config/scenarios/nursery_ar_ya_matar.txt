Rain physics
// Raindrop falls with slight angle (wind effect)
const windAngle = 5; // degrees
const fallSpeed = 8; // px per frame
const x = startX + frame * Math.tan(windAngle * Math.PI/180) * fallSpeed;
const y = startY + frame * fallSpeed;


// Splash when hitting ground
const splashFrame = (groundY - startY) / fallSpeed;
const isSplashing = frame > splashFrame;
