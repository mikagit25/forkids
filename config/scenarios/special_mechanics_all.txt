// Прятки — механика появления/исчезновения
const peekAnimation = {
  phases: ['hidden', 'peek', 'show', 'hide'],
  phaseDuration: [60, 45, 90, 45], // кадры


  peek: (frame, hideX, showX) => {
    // Персонаж выезжает частично из-за укрытия
    const peekProgress = interpolate(frame, [0, 45], [0, 1], {
      easing: Easing.out(Easing.cubic)
    });
    return hideX + (showX - hideX) * 0.4 * peekProgress; // только 40%
  },
};

// Тень от объекта
const shadowProject = (obj, lightX, lightY, wallX) => {
  // Параметрическое уравнение луча от источника через объект до стены
  const t = (wallX - lightX) / (obj.x - lightX);
  return {
    x: wallX,
    y: lightY + (obj.y - lightY) * t,
    scale: t, // тень больше чем объект
  };
};


Блок 2 (5–10 мин): ЖИВОТНЫЕ-ТЕНИ Серия силуэтов:

СЕРИЯ 11 — ЧИСЛА БЕЗ СЛОВ (Silent Count)
COUNT-A — БЕЗ СЛОВ (серия 10 видео × 20 мин)
Концепция: Предметы появляются по одному. Цифра растёт. Никаких слов — только визуальный счёт. Satisfying.
Видео 1 — ОДНО (1)
* Одно яблоко медленно опускается с неба.
* Цифра 1 появляется — золотая.
* Яблоко покачивается. Цифра пульсирует.
* Повтор с разными предметами: 1 мяч, 1 кошка, 1 луна.
Видео 2 — ДВА (2)
* Первое яблоко. Цифра 1.
* Второе яблоко присоединяется. Цифра плавно меняется 1 → 2.
* Оба прыгают вместе. Цифра 2 светится.
(Видео 3-10 — аналогично до числа 10)
Ключевая механика:
// Число меняется при добавлении предмета
const countTransition = (frame, prevCount, newCount, changeFrame) => {
  if (frame < changeFrame) return prevCount;
  const progress = Math.min((frame - changeFrame) / 30, 1);
  // Число "прыгает" при смене: scale 1 → 1.5 → 1
  const scale = 1 + Math.sin(progress * Math.PI) * 0.5;
  return { count: newCount, scale };
};

// ПРЯТКИ — появление из-за объекта
const peekFromBehind = (frame, hideX, obj, blocker) => {
  const peek = Math.max(0, obj.x - blocker.rightEdge);
  const visibility = peek / obj.width; // 0=спрятан, 1=виден полностью
  return { x: Math.max(blocker.rightEdge - obj.width * (1-visibility), hideX) };
};


// ТЕНИ — правильная проекция
const castShadow = (obj, light, ground) => {
  const t = (ground.y - light.y) / (obj.y - light.y);
  return {
    x: light.x + (obj.x - light.x) * t,
    scaleX: t,
    scaleY: 0.1, // тень плоская
    opacity: interpolate(t, [1, 5], [0.7, 0.2]),
  };
};


// ПУЗЫРЬ — переливы (iridescence)
const iridescent = (frame, x, y, radius) => {
  return Array.from({length: 5}).map((_, i) => ({
    hue: (frame * 1.5 + i * 60 + x * 0.1 + y * 0.1) % 360,
    opacity: 0.08,
    offsetX: Math.sin(frame * 0.03 + i) * 3,
    offsetY: Math.cos(frame * 0.04 + i) * 3,
  }));
};


// ЧИСЛА — появление с эффектом
const numberReveal = (frame, count, prevCount, triggerFrame) => {
  if (frame < triggerFrame) return prevCount;
  const age = frame - triggerFrame;
  const scale = 1 + Math.sin(age * 0.3) * 0.4 * Math.exp(-age * 0.05);
  return { count, scale, glow: Math.exp(-age * 0.03) };
};
