СЕРИЯ 15 — СТРОИТЕЛЬСТВО (Construction / البناء)
Концепция
Что-то строится медленно — кирпич за кирпичом, деталь за деталью. Самый satisfying детский контент — порядок возникает из хаоса.
________________


BUILD1-A — СТРОИМ ДОМ (без слов, 30 мин)
BPM: 68 | Satisfying нарастающий
Фазы строительства:
Фаза 1 (0–6 мин): ФУНДАМЕНТ
* Пустое поле. Экскаватор (камео!) копает яму.
* Бетон заливается: серая жидкость медленно заполняет котлован.
* Застывает — становится светлее.
* Фундамент готов. Маленький флажок.
Фаза 2 (6–12 мин): СТЕНЫ — КИРПИЧ ЗА КИРПИЧОМ Это самый satisfying блок всего видео.
// Кирпичная кладка
interface Brick { row: number; col: number; color: string; }


const WALL_CONFIG = { rows: 8, cols: 12, brickW: 60, brickH: 25 };


// Кирпичи появляются по одному — слева направо, снизу вверх
const brickOrder = generateBrickOrder(WALL_CONFIG);
// Каждый кирпич: падает сверху на своё место
const getBrickY = (frame, brick, brickIndex) => {
  const startFrame = brickIndex * 12; // 12 кадров между кирпичами
  if (frame < startFrame) return -100; // не видно
  const age = frame - startFrame;
  return interpolate(age, [0, 15], [brick.finalY - 80, brick.finalY], {
    easing: Easing.out(Easing.bounce),
    extrapolateRight: 'clamp',
  });
};


Фаза 3 (12–18 мин): КРЫША
* Балки укладываются одна за другой.
* Черепица — каждая плитка появляется по очереди.
* Конёк — последний штрих.
Фаза 4 (18–24 мин): ОКНА И ДВЕРИ
* Дверь: рама → дверное полотно → ручка.
* Окна: рама → стёкла → ставни.
* Каждый элемент с отдельной satisfying анимацией.
Фаза 5 (24–30 мин): ДОМ ГОТОВ
* Финальный вид: красивый готовый дом.
* Цветы вырастают у входа (связь с садом).
* Кошка садится у двери (камео!).
* Все персонажи из всех серий подходят посмотреть.
* Флаг на крыше поднимается — ветер колышет.

REMOTION — ФИНАЛЬНЫЕ КОМПОНЕНТЫ
// КИРПИЧНАЯ КЛАДКА — главный компонент серии строительства
const BRICK_COLORS = { red: '#CC3B2A', tan: '#D4A574', gray: '#9A9A9A' };


const Brick = ({ row, col, config, frame, appearFrame }) => {
  const age = frame - appearFrame;
  if (age < 0) return null;


  const finalY = config.groundY - row * (config.brickH + config.mortarH);
  const currentY = interpolate(age, [0, 20], [finalY - 60, finalY], {
    easing: Easing.out(Easing.bounce),
    extrapolateRight: 'clamp',
  });
  // Лёгкое покачивание при приземлении
  const wobble = age < 25 ? Math.sin(age * 0.5) * (1 - age/25) * 3 : 0;


  return (
    <rect
      x={config.groundX + col * (config.brickW + config.mortarW) + wobble}
      y={currentY}
      width={config.brickW}
      height={config.brickH}
      fill={BRICK_COLORS.red}
      rx={2}
    />
  );
};


// НОТА из инструмента — вылетает и тает
const MusicNote = ({ x, y, frame, birthFrame, direction }) => {
  const age = frame - birthFrame;
  if (age < 0 || age > 90) return null;


  const floatY = y - age * 1.2;
  const floatX = x + Math.sin(age * 0.1) * 20 * direction;
  const opacity = interpolate(age, [0, 20, 80, 90], [0, 1, 1, 0]);
  const scale = interpolate(age, [0, 10], [0.5, 1], { extrapolateRight: 'clamp' });


  return (
    <g transform={`translate(${floatX}, ${floatY}) scale(${scale})`} opacity={opacity}>
      <NoteShape /> {/* SVG нота */}
    </g>
  );
};


// АККОРДЕОН — расширение-сжатие
const Accordion = ({ frame }) => {
  const cycle = (frame % (BREATH_PERIOD * 30)) / (BREATH_PERIOD * 30);
  const expand = Math.sin(cycle * 2 * Math.PI) * 0.5 + 0.5;
  const bellowsWidth = lerp(MIN_WIDTH, MAX_WIDTH, expand);
  // Рисуем зигзаг меха между двумя панелями
  const zigzagPoints = generateBellows(bellowsWidth);
  return <polyline points={zigzagPoints} stroke="#8B4513" strokeWidth={3} fill="none"/>;
};


// ЗАМОК — масштабное строительство
const CASTLE_ELEMENTS = [
  { type: 'wall', x: 100, y: 500, w: 700, h: 200, delay: 0 },
  { type: 'tower', x: 50, y: 200, w: 120, h: 350, delay: 150 },
  { type: 'tower', x: 730, y: 200, w: 120, h: 350, delay: 200 },
  { type: 'battlement', count: 8, y: 200, delay: 300 },
  { type: 'gate', x: 300, y: 350, w: 200, h: 200, delay: 400 },
  { type: 'drawbridge', x: 300, y: 540, delay: 500 },
];


// Все элементы замка появляются в порядке задержки
CASTLE_ELEMENTS.map(el => {
  const visible = frame > el.delay;
  const buildProgress = visible
    ? Math.min((frame - el.delay) / 60, 1)
    : 0;
  // Элемент "вырастает" снизу вверх
  return { ...el, clipHeight: buildProgress * el.h };
});
